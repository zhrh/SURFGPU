75 void Kmeans::Run(IpVec *ipts, int clusters, bool init)
76 {
77  if (!ipts->size()) return;
78 
79  SetIpoints(ipts);
80 
81  if (init) InitRandomClusters(clusters);
82 
83  while (AssignToClusters());
84  {
85  RepositionClusters();
86  }
87 }

98 void Kmeans::InitRandomClusters(int n)
99 {
100  // clear the cluster vector
101  clusters.clear();
102 
103  // Seed the random number generator
104  srand((int)time(NULL));
105 
106  // add 'n' random ipoints to clusters list as initial centers
107  for (int i = 0; i < n; ++i)
108  {
109  clusters.push_back(ipts->at(rand() % ipts->size()));
110  }
111 }

115 bool Kmeans::AssignToClusters()
116 {
117  bool Updated = false;
118 
119  // loop over all Ipoints and assign each to closest cluster
120  for (unsigned int i = 0; i < ipts->size(); ++i)
121  {
122  float bestDist = FLT_MAX;
123  int oldIndex = ipts->at(i).clusterIndex;
124 
125  for (unsigned int j = 0; j < clusters.size(); ++j)
126  {
127  float currentDist = Distance(ipts->at(i), clusters[j]);
128  if (currentDist < bestDist)
129  {
130  bestDist = currentDist;
131  ipts->at(i).clusterIndex = j;
132  }
133  }
134 
135  // determine whether point has changed cluster
136  if (ipts->at(i).clusterIndex != oldIndex) Updated = true;
137  }
138 
139  return Updated;
140 }

144 void Kmeans::RepositionClusters()
145 {
146  float x, y, dx, dy, count;
147 
148  for (unsigned int i = 0; i < clusters.size(); ++i)
149  {
150  x = y = dx = dy = 0;
151  count = 1;
152 
153  for (unsigned int j = 0; j < ipts->size(); ++j)
154  {
155  if (ipts->at(j).clusterIndex == i)
156  {
157  Ipoint ip = ipts->at(j);
158  x += ip.x;
159  y += ip.y;
160  dx += ip.dx;
161  dy += ip.dy;
162  ++count;
163  }
164  }
165 
166  clusters[i].x = x/count;
167  clusters[i].y = y/count;
168  clusters[i].dx = dx/count;
169  clusters[i].dy = dy/count;
170  }
171 }

180  + pow(ip1.dy - ip2.dy, 2)*/);
181 }
