37 inline void surfDetDes(IplImage *img, /* image to find Ipoints in */
38  std::vector<Ipoint> &ipts, /* reference to vector of Ipoints */
39  bool upright = false, /* run in rotation invariant mode? */
40  int octaves = OCTAVES, /* number of octaves to calculate */
41  int intervals = INTERVALS, /* number of intervals per octave */
42  int init_sample = INIT_SAMPLE, /* initial sampling step */
43  float thres = THRES /* blob response threshold */)
44 {
45 
46  struct timeval start,end;
47  int time_use;
48  gettimeofday(&start,NULL);
49  // Create integral-image representation of the image
50  cudaImage *int_img = Integral(img);
51  gettimeofday(&end,NULL);
52  time_use = 1000000 * (end.tv_sec - start.tv_sec) + end.tv_usec - start.tv_usec;
53  time_use /= 1000;
54  std::cout<<"Create integral-image time is:"<<time_use<<"ms"<<std::endl;
55 
56  gettimeofday(&start,NULL);
57  // Create Fast Hessian Object
58  FastHessian fh(int_img, ipts, octaves, intervals, init_sample, thres);
59 
60  // Extract interest points and store in vector ipts
61  fh.getIpoints();
62  gettimeofday(&end,NULL);
63  time_use = 1000000 * (end.tv_sec - start.tv_sec) + end.tv_usec - start.tv_usec;
64  time_use /= 1000;
65  std::cout<<"Extract interest points time is:"<<time_use<<"ms"<<std::endl;
66 
67  gettimeofday(&start,NULL);
68  if (ipts.size() > 0)
69  {
70  // Create Surf Descriptor Object
71  Surf des(int_img, ipts);
72 
73  // Extract the descriptors for the ipts
74  des.getDescriptors(upright);
75  }
76  gettimeofday(&end,NULL);
77  time_use = 1000000 * (end.tv_sec - start.tv_sec) + end.tv_usec - start.tv_usec;
78  time_use /= 1000;
79  std::cout<<"Extarc descriptors time is:"<<time_use<<"ms"<<std::endl;
80 
81  // Deallocate the integral image
82  freeCudaImage(int_img);
83 }
